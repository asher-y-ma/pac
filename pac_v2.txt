function FindProxyForURL(url, host)
{
    url  = url.toLowerCase();
    host = host.toLowerCase();

    if (shExpMatch(url,"*googleusercontent.com")
      || shExpMatch(url,"*gvt1.com")
      || shExpMatch(url,"*myip.top")
    ) {
       return "PROXY 127.0.0.1:8888";
    }
    return "PROXY 127.0.0.1:9999";
}
